26  package agave;
27  
28  import agave.internal.DestinationImpl;
29  
30  /**
31   * A factory class for creating {@link Destination}s.
32   * 
33   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
34   * 
35   * @see Destination
36   */
37  public final class Destinations {
38  
39  	/**
40  	 * Creates a new {@link Destination} within the deployed context.  The destination will be 
41       * redirected to if the HTTP request method was a POST, otherwise it will be forwarded to.
42  	 * 
43  	 * @param path the desired destination path relative to the deployed context
44  	 * @return a constructed {@link Destination}
45  	 */
46  	public static Destination create(String path) {
47  		return new DestinationImpl(path);
48  	}
49  	
50  	/**
51  	 * Creates a new {@link Destination} within the deployed context that will be redirected to. 
52  	 * 
53  	 * @param path the desired destination path relative to the deployed context (eg: {@code /section/resource}}
54  	 * @return a constructed {@link Destination}
55  	 */
56  	public static Destination redirect(String path) {
57  		return new DestinationImpl(path, true);
58  	}
59  	
60  	/**
61  	 * Creates a new {@link Destination} within the deployed context that will be forwarded to.
62  	 * 
63  	 * @param path the desired resource relative to the context path to forward to (eg: {@code /WEB-INF/jsp/index.jsp}) 
64  	 * @return a constructed {@link Destination}
65  	 */
66  	public static Destination forward(String path) {
67  		return new DestinationImpl(path, false);
68  	}
69  	
70  }
