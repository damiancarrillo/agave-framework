37  /**
38   * A {@code URIPattern} is the object that indicates which handler should be
39   * invoked according to the requested URI. A {@code URIPattern} is similar in
40   * nature to the string part of the URI except for having wildcards and
41   * replacement variables.
42   * 
43   * Replacement variables look like {@code ${var}} and are supplied to handler
44   * methods as arguments to the method if annotated. From this point of view,
45   * though, consider replacement variables as a single wildcard match.
46   * 
47   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
48   */
49  public class URIPatternImpl implements URIPattern {
50      
51      private static final String FORWARD_SLASH = "/";
52      private static final Pattern replacementPattern = Pattern.compile("\\$\\{(.*)\\}");
53  
54      private String pattern;
55      private String[] parts;
56  
57      public URIPatternImpl(String pattern) {
58          if (!pattern.startsWith("/")) {
59              throw new IllegalArgumentException(
60                  "The supplied pattern must begin with a forward slash (\"/\") "
61                      + "where the root is relative to the context path.");
62          }
63          if (pattern.contains("**/${")) {
64              throw new IllegalArgumentException(
65                  "The supplied pattern is nondeterministic.  There is no way of "
66                      + "knowing when to stop matching with this type of pattern: /**/${var}/");
67          }
68          this.pattern = normalizePattern(pattern);
69          if (pattern.length() > 1) {
70              this.parts = pattern.substring(1).split("/");
71          }
72      }
73  
74      public String getPattern() {
75          return pattern;
76      }

78      protected String normalizePattern(String pattern) {
79          URI uri;
80          try {
81              // replace illegal characters in the pattern so that URI can do the
82              // work of normalizing the uri - hopefully nobody chooses these
83              // strings in the url :-P
84              pattern = pattern.replace("${", "~~agave~~start~~delim~~");
85              pattern = pattern.replace("}", "~~agave~~end~~delim~~");
86              uri = new URI(pattern);
87          } catch (URISyntaxException ex) {
88              throw new IllegalArgumentException("Malformed URI pattern: " + pattern, ex);
89          }
90          String normalizedUri = stripTrailingSlash(uri.normalize().toString());
91          normalizedUri = normalizedUri.replace("~~agave~~start~~delim~~", "${");
92          normalizedUri = normalizedUri.replace("~~agave~~end~~delim~~", "}");
93          normalizedUri = condenseWildcards(normalizedUri);
94          return normalizedUri;
95      }
96  
97      /**
98       * Condenses multiple successive wildcards into the most generic wildcard
99       */
100     private String condenseWildcards(String pattern) {
101         while (pattern.contains("**/**") || pattern.contains("**/*")
102             || pattern.contains("*/**")) {
103             pattern = pattern.replace("**/**", "**");
104             pattern = pattern.replace("**/*", "**");
105             pattern = pattern.replace("*/**", "**");
106         }
107         return pattern;
108     }

110     /**
111      * Normalizes the URI string so that .. and . are properly handled and
112      * condensed.
113      * 
114      * @param uriStr
115      *            the URI string to normalize
116      * @return the normalized URI string
117      */
118     public String normalizeURI(String uriStr) {
119         URI uri;
120         try {
121             uri = new URI(uriStr);
122         } catch (URISyntaxException ex) {
123             throw new IllegalArgumentException("Malformed URI: " + uriStr, ex);
124         }
125         return stripTrailingSlash(uri.normalize().toString());
126     }
127 
128     private String stripTrailingSlash(String input) {
129         if (!input.equals(FORWARD_SLASH) && input.endsWith(FORWARD_SLASH)) {
130             return input.substring(0, input.length() - 1);
131         }
132         return input;
133     }

135     /**
136      * Determines whether the supplied URI string matches the pattern that this
137      * {@code URIPattern} encapsulates. The URI string supplied as an argument
138      * must start with a forward slash ('/'). The URI string is normalized with
139      * URI.normalize() from the Java API, then compared against the stored
140      * pattern where wildcards and replacement variables help determine the
141      * match. Replacement variables look like <code>${someVar}</code> and are
142      * taken as an automatic match. A single asterisk represents a wildcard
143      * match where the supplied token matches automatically as well. A double
144      * asterisk matches multiple tokens until the next token in the pattern is
145      * matched against the URI.
146      * @param uri the uri string
147      * @return true if the uri matches this pattern
148      */
149     public boolean matches(String uri) {
150         if (!uri.startsWith(FORWARD_SLASH)) {
151             throw new IllegalArgumentException("URI must begin with a forward slash ('/')");
152         }
153 
154         // check for root (so as to not match all patterns starting with a '/')
155         if ("/".equals(uri)) {
156             if ("/".equals(pattern)) {
157                 return true;
158             }
159             return false;
160         }
161         
162         String[] patternTokens = pattern.split(FORWARD_SLASH);
163         String[] uriTokens = normalizeURI(uri).split(FORWARD_SLASH);
164         
165         int pi = 0, ui = 0;
166         for (; pi < patternTokens.length && ui < uriTokens.length; pi++, ui++) {
167             if ("**".equals(patternTokens[pi])) {
168                 ++pi;
169                 if (pi >= patternTokens.length) {
170                     // matches the rest of the uri
171                     return true;
172                 } else {
173                     // slurp the uri tokens until they match the next position
174                     // in the pattern
175                     while (ui < uriTokens.length) {
176                         if (uriTokens[ui].equalsIgnoreCase(patternTokens[pi])) {
177                             break;
178                         }
179                         ++ui;
180                     }
181                     if (ui >= uriTokens.length) {
182                         return false;
183                     }
184                 }
185             } else if (!uriTokens[ui].equalsIgnoreCase(patternTokens[pi])
186                 && !"*".equals(patternTokens[pi])
187                 && !Pattern.compile("\\$\\{.*\\}").matcher(patternTokens[pi]).matches()) {
188                 return false;
189             }
190         }
191 
192         if ((pi >= patternTokens.length && ui < uriTokens.length)
193             || (ui < uriTokens.length && !"**".equals(patternTokens[pi]))
194             || (pi < patternTokens.length && !"**".equals(patternTokens[pi]))) {
195             return false;
196         }
197 
198         return true;
199     }

216     /**
217      * Compares two {@code URIPattern}s for greater specificity. A more
218      * specific {@code URIPattern} should always be sorted before a more generic
219      * one.
220      * 
221      * @param that
222      *            the {@code URIPattern} to compare against
223      * @return -1 if this {@code URIPattern} is more specific, 0 if they are
224      *         equal in specificity and 1 if that {@code URIPattern} is more
225      *         specific
226      */
227     public int compareTo(URIPattern that) {
228         if (this.equals(that)) {
229             return 0;
230         }
231 
232         Integer value = null;
233 
234         String[] thisTokens = pattern.split(FORWARD_SLASH);
235         String[] thatTokens = that.getPattern().split(FORWARD_SLASH);
236 
237         for (int i = 0; i < thisTokens.length && i < thatTokens.length; i++) {
238             if ("**".equals(thisTokens[i]) && !"**".equals(thatTokens[i])) {
239                 value = 1;
240                 break;
241             } else if (!"**".equals(thisTokens[i]) && "**".equals(thatTokens[i])) {
242                 value = -1;
243                 break;
244             } else if ("*".equals(thisTokens[i]) && !"*".equals(thatTokens[i])) {
245                 value = 1;
246                 break;
247             } else if (!"*".equals(thisTokens[i]) && "*".equals(thatTokens[i])) {
248                 value = -1;
249                 break;
250             }
251         }
252         
253         if (value == null) {
254             if (thisTokens.length > thatTokens.length) {
255                 value = -1;
256             } else if (thisTokens.length < thatTokens.length) {
257                 value = 1;
258             } else {
259                 int length = (thisTokens.length > thatTokens.length) ? thisTokens.length : thatTokens.length;
260                 for (int i = 0; i < length; i++) {
261                     Matcher thisTokenMatcher = replacementPattern.matcher(thisTokens[i]);
262                     Matcher thatTokenMatcher = replacementPattern.matcher(thatTokens[i]);
263                     
264                     if (thisTokenMatcher.matches() && thatTokenMatcher.matches()) {
265                         value = 0;
266                         continue; // just ignore this case - treat all replacement params equal
267                     } else if (!thisTokenMatcher.matches() && thatTokenMatcher.matches()) {
268                         value = -1;
269                         break;
270                     } else if (thisTokenMatcher.matches() && !thatTokenMatcher.matches()) {
271                         value = 1;
272                         break;
273                     }
274                     
275                     value = thisTokens[i].compareToIgnoreCase(thatTokens[i]);
276                     if (value != 0) {
277                         break;
278                     }
279                 }
280             }
281         }
282 
283         return value;
284     }

286     public Map<String, String> getParameterMap(HttpServletRequest request) {
287         String uri = request.getServletPath();
288         Map<String, String> parameterMap = new HashMap<String, String>();
289         if (parts != null && uri != null && uri.length() > 1) {
290             String[] requestedParts = uri.substring(1).split("/");
291             if (requestedParts.length >= parts.length) {
292                 for (int i = 0; i < parts.length; i++) {
293                     Matcher matcher = replacementPattern.matcher(parts[i]);
294                     if (matcher.matches() && matcher.groupCount() > 0) {
295                         String paramName = matcher.group(1);
296                         String paramValue = requestedParts[i]; 
297                         parameterMap.put(paramName, paramValue);
298                     }
299                 }
300             }
301         }
302         return parameterMap;           
303     }
