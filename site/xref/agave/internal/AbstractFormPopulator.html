44  public abstract class AbstractFormPopulator implements FormPopulator {
45  
46  	private static final String ILLEGAL_ARGUMENT_EXCEPTION_MSG = 
47  		"Mutator {0}#{1}(...) is expecting argument of type {2} and recieved {3}";
48  
49      protected SortedMap<String, List<Object>> parameters = new TreeMap<String, List<Object>>();
50      protected Locale locale;
51  
52      protected AbstractFormPopulator(Locale locale) {
53          this.locale = locale;
54      }

56      public SortedMap<String, List<Object>> getParameters() {
57          return parameters;
58      }

60      public void populate(Object formInstance) 
61          throws NoSuchMethodException, 
62                 SecurityException, 
63                 IllegalAccessException, 
64                 IllegalArgumentException, 
65                 InvocationTargetException,
66                 InstantiationException,
67                 ConversionException {
68          CallChain callChain = null;
69          for (String parameterName : parameters.keySet()) {
70              List<Object> parameterValues = parameters.get(parameterName);
71              boolean unique = true;
72              if (parameterValues != null && parameterValues.size() > 1) {
73                  unique = false;
74              }
75              callChain = new CallChainImpl(parameterName, unique);
76              populateProperty(formInstance, callChain, parameterValues);
77          }
78      }

80      private void populateProperty(Object formInstance, CallChain callChain, List<Object> parameterValues) 
81      throws NoSuchMethodException, 
82             SecurityException, 
83             IllegalAccessException, 
84             IllegalArgumentException, 
85             InvocationTargetException,
86             InstantiationException,
87             ConversionException {
88          Object targetInstance = formInstance;
89          Class<?> targetClass = targetInstance.getClass();
90      
91          for (String accessorName : callChain.getAccessorNames()) {
92              try {
93                  Method accessor = targetClass.getMethod(accessorName);
94                  targetInstance = accessor.invoke(targetInstance);
95                  targetClass = targetInstance.getClass();
96              } catch (NoSuchMethodException ex) {
97                  throw new NoSuchMethodException("Missing accessor \"" + accessorName + "\" on " + targetClass.getName() 
98                      + " invoked through request parameter \"" + callChain.getParameterName() + "\"");
99              }
100         }
101     
102         for (Method mutator : targetClass.getMethods()) {
103             if (mutator.getName().equals(callChain.getMutatorName())) {
104                 Object parameterValue = null;
105                 switch (callChain.getMutatorType()) {
106                     case SETTING:
107                         if (parameterValues != null && !parameterValues.isEmpty()) {
108                             parameterValue = parameterValues.get(0);
109                         }
110                         setOrAppendProperty(mutator, targetInstance, parameterValue);
111                         break;
112                     case APPENDING:
113                         for (Object param: parameterValues) {
114                             setOrAppendProperty(mutator, targetInstance, param);
115                         }
116                         break;
117                     case INSERTING:
118                         if (parameterValues != null && !parameterValues.isEmpty()) {
119                             parameterValue = parameterValues.get(0);
120                         }
121                         insertProperty(mutator, targetInstance, callChain.getIndex(), parameterValue);
122                         break;
123                     case PUTTING:
124                         if (parameterValues != null && !parameterValues.isEmpty()) {
125                             parameterValue = parameterValues.get(0);
126                         }
127                         putProperty(mutator, targetInstance, callChain.getKey(), parameterValue);
128                         break;
129                 }
130             }
131         }
132     }
133     
134     private void setOrAppendProperty(Method mutator, Object targetInstance, Object parameterValue) 
135         throws IllegalAccessException, 
136                IllegalArgumentException, 
137                InvocationTargetException, 
138                InstantiationException,
139                ConversionException {
140 		try {
141 	        mutator.invoke(targetInstance, convertIfNecessary(mutator, parameterValue));
142 		} catch (IllegalArgumentException ex) {
143 			String errorMessage = MessageFormat.format(ILLEGAL_ARGUMENT_EXCEPTION_MSG, 
144 				mutator.getDeclaringClass().getName(), 
145 				mutator.getName(), 
146 				mutator.getParameterTypes()[0].getName(), 
147 				parameterValue.getClass().getName());
148 			throw new IllegalArgumentException(errorMessage);
149 		}
150     }
151 
152     private void insertProperty(Method mutator, Object targetInstance, int index, Object parameterValue)
153         throws IllegalAccessException, 
154                IllegalArgumentException, 
155                InvocationTargetException, 
156                InstantiationException,
157                ConversionException {
158         mutator.invoke(targetInstance, index, convertIfNecessary(mutator, parameterValue));
159     }
160     
161     private void putProperty(Method mutator, Object targetInstance, String key, Object parameterValue)
162     throws IllegalAccessException, 
163            IllegalArgumentException, 
164            InvocationTargetException, 
165            InstantiationException,
166            ConversionException {
167         mutator.invoke(targetInstance, key, convertIfNecessary(mutator, parameterValue));
168     }

170     @SuppressWarnings("unchecked")
171     private Object convertIfNecessary(Method mutator, Object parameterValue) 
172     throws ConversionException, 
173            InstantiationException,
174            IllegalAccessException {
175         Class<?>[] parameterTypes = mutator.getParameterTypes();
176         
177         if (parameterTypes != null) {
178             int parameterOffset = (parameterTypes.length == 1) ? 0 : 1;
179             Class<?> parameterType = parameterTypes[parameterOffset];
180             Converter converter = null; // keep this vague 
181             // first look for a ConvertWith annotation
182             for (Annotation annotation : mutator.getParameterAnnotations()[parameterOffset]) {
183                 if (annotation instanceof ConvertWith) {
184                     converter = ((ConvertWith)annotation).value().newInstance();
185                     break;
186                 }
187             }
188 
189             // try to look up a converter for common types
190             if (converter == null) {
191                 if (parameterType.isAssignableFrom(Boolean.class) || parameterType.isAssignableFrom(boolean.class)) {
192                     converter = new BooleanConverter();
193                 } else if (parameterType.isAssignableFrom(Byte.class) || parameterType.isAssignableFrom(byte.class)) {
194                     converter = new ByteConverter();
195                 } else if (parameterType.isAssignableFrom(Character.class) || parameterType.isAssignableFrom(char.class)) {
196                     converter = new CharacterConverter();
197                 } else if (parameterType.isAssignableFrom(Double.class) || parameterType.isAssignableFrom(double.class)) {
198                     converter = new DoubleConverter();
199                 } else if (parameterType.isAssignableFrom(Float.class) || parameterType.isAssignableFrom(float.class)) {
200                     converter = new FloatConverter();
201                 } else if (parameterType.isAssignableFrom(Integer.class) || parameterType.isAssignableFrom(int.class)) {
202                     converter = new IntegerConverter();
203                 } else if (parameterType.isAssignableFrom(Long.class) || parameterType.isAssignableFrom(long.class)) {
204                     converter = new LongConverter();
205                 } 
206             }
207             
208             if (converter != null) {
209                 return converter.convert(parameterValue, locale);
210             }
211         }
212         
213         return parameterValue;
214     }
