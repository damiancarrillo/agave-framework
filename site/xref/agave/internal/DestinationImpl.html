26  package agave.internal;
27  
28  import java.util.ArrayList;
29  import java.util.Collections;
30  import java.util.List;
31  import java.util.Map;
32  import java.util.TreeMap;
33  
34  import javax.servlet.ServletContext;
35  
36  import agave.Destination;
37  
38  /**
39   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
40   */
41  public class DestinationImpl implements Destination {
42      
43      public static final String ENCODED_AMPERSAND = "&amp;";
44  
45      private String path;
46      private Map<String, List<String>> parameters = new TreeMap<String, List<String>>();
47      private Boolean redirect;
48      
49      /**
50       * Create a new {@code Destination} within the deployed context.  The destination will be 
51       * redirected to if the HTTP request method was a POST, otherwise it will be forwarded to.
52       * @param path
53       */
54      public DestinationImpl(String path) {
55          setPath(path);
56      }
57      
58      public DestinationImpl(String path, boolean redirect) {
59          setPath(path);
60          setRedirect(Boolean.valueOf(redirect));
61      }
62      
63      public void addParameter(String name, String value) {
64          if (!parameters.containsKey(name)) {
65              parameters.put(name, new ArrayList<String>());
66          }
67          parameters.get(name).add(value);
68      }
69      
70      public String getPath() {
71          return path;
72      }
73  
74      public void setPath(String path) {
75          if (!path.startsWith("/")) {
76              throw new IllegalArgumentException("Destination paths should start with a forward slash ('/')"
77                      + " signifying the root of the context");
78          }
79          this.path = path;
80      }
81  
82      public Map<String, List<String>> getParameters() {
83          return parameters;
84      }
85  
86      public void setParameters(Map<String, List<String>> parameters) {
87          this.parameters = parameters;
88      }
89  
90      public Boolean getRedirect() {
91          return redirect;
92      }
93      
94      public void setRedirect(Boolean redirect) {
95          this.redirect = redirect;
96      }
97      
98      public String encode(ServletContext context) {
99          StringBuilder path = new StringBuilder();
100         if (getPath() != null) {
101             path.append(getPath());
102         }
103         if (!getParameters().isEmpty()) {
104             path.append("?");
105             boolean first = true;
106             for (String parameterName : getParameters().keySet()) {
107                 Collections.sort(getParameters().get(parameterName));
108                 for (String parameterValue : getParameters().get(parameterName)) {
109                     if (!first) {
110                         path.append("&");
111                     }
112                     if (parameterValue.contains("&")) {
113                         parameterValue = parameterValue.replace("&", ENCODED_AMPERSAND);
114                     }
115                     path.append(parameterName).append("=").append(parameterValue);
116                     first = false;
117                 }
118             }
119         }
120         return path.toString();
121     }
