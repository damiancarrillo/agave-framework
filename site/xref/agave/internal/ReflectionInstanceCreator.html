26  package agave.internal;
27  
28  import agave.InstanceCreator;
29  import agave.exception.FormException;
30  import agave.exception.HandlerException;
31  
32  
33  /**
34   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
35   */
36  public class ReflectionInstanceCreator implements InstanceCreator {
37  
38      /**
39       * Creates a new instance of a form object for the form class specified in the supplied descriptor by
40       * calling its default constructor.
41       * @param descriptor the handler descriptor that describes which form to instantiate
42       * @throws IllegalAccessException 
43       * @throws InstantiationException 
44       * @throws FormError when a form instance failed to be instantiated
45       */
46      public Object createFormInstance(HandlerDescriptor descriptor) throws FormException {
47          Object formInstance = null;
48          if (descriptor.getFormClass() != null) {
49              try {
50                  formInstance = descriptor.getFormClass().newInstance();
51              } catch (InstantiationException ex) {
52                  throw new FormException(descriptor, ex);
53              } catch (IllegalAccessException ex) {
54                  throw new FormException(descriptor, ex);
55              }
56          }
57          return formInstance;
58      }
59  
60  
61      /**
62       * Creates a new instance of a handler object for the handler class specified in the supplied descriptor 
63       * by calling its default constructor.
64       * @param descriptor the handler descriptor that describes which form to instantiate.
65       * @throws FormError when a handler instance failed to be instantiated
66       */
67      public Object createHandlerInstance(HandlerDescriptor descriptor) throws HandlerException {
68          Object handlerInstance = null;
69          if (descriptor.getHandlerClass() != null) {
70              try {
71                  handlerInstance = descriptor.getHandlerClass().newInstance();
72              } catch (InstantiationException ex) {
73                  throw new HandlerException(descriptor, ex);
74              } catch (IllegalAccessException ex) {
75                  throw new HandlerException(descriptor, ex);
76              }
77          }
78          return handlerInstance;
79      }

83      }
