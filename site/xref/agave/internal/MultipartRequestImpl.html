46  public class MultipartRequestImpl extends HttpServletRequestWrapper implements MultipartRequest {

52      public MultipartRequestImpl(HttpServletRequest request) throws IOException {
53          super(request);
54          try {
55              parser = new MultipartParserImpl(request);
56              parser.parseInput();
57          } finally {
58              request.getInputStream().close();
59          }
60      }

62      /**
63       * Gets a parameter from the query string or the post data in that order.  In the case of multiple
64       * values for a single parameter, the first value is taken, and this initial value will always be 
65       * the same.
66       * @see #getParameterValues(String) How to access the other values of a multi-valued parameter
67       */
68      @Override
69      public String getParameter(String name) {
70          if (getParameters().get(name) != null) {
71              for (String value : getParameters().get(name)) {
72                  return value;
73              }
74          }
75          return super.getParameter(name);
76      }
77  
78      /**
79       * Returns an immutable {@code Map} of parameter names and their values.  There can be multiple
80       * values for a single parameter whenever the same parameter name is supplied multiple times.  An 
81       * example of this would be a URL with a query string like {@code ?catGoes=meow&catGoes=hiss} or
82       * a form with multiple inputs that use the same {@code name}.
83       */
84      @Override
85      public Map<String, String[]> getParameterMap() {
86          if (this.parameterMap == null) {
87              this.parameterMap = new HashMap<String, String[]>();
88              for (Object parameterNameObj : super.getParameterMap().keySet()) {
89                  String parameterName = parameterNameObj.toString();
90                  Collection<String> parameterValues = new ArrayList<String>();
91                  if (super.getParameterMap().get(parameterName) != null) {
92                      String[] superParameterValues = (String[])super.getParameterMap().get(parameterName);
93                      for (String parameterValue : superParameterValues) {
94                          parameterValues.add(parameterValue);
95                      }
96                  }
97                  if (getParameters().get(parameterName) != null) {
98                      for (String parameterValue : getParameters().get(parameterName)) {
99                          parameterValues.add(parameterValue);
100                     }
101                 }
102                 this.parameterMap.put(parameterName, parameterValues.toArray(new String[] {}));
103             }
104             for (String parameterName : getParameters().keySet()) {
105                 if (!this.parameterMap.containsKey(parameterName)) {
106                     this.parameterMap.put(parameterName, getParameters().get(parameterName).toArray(new String[] {}));
107                 }
108             }
109             this.parameterMap = Collections.unmodifiableMap(this.parameterMap);
110         }
111         return this.parameterMap;
112     }

114     @SuppressWarnings("unchecked")
115 	@Override
116     public Enumeration<String> getParameterNames() {
117         Vector<String> parameterNames = new Vector<String>();
118         parameterNames.addAll(super.getParameterMap().keySet());
119         parameterNames.addAll(getParameters().keySet());
120         return parameterNames.elements();
121     }

123     @Override
124     public String[] getParameterValues(String name) {
125         return getParameterMap().get(name);

136     public static boolean isMultipart(HttpServletRequest request) {
137     	return request.getContentType() != null && request.getContentType().startsWith("multipart/form-data");
138     }
