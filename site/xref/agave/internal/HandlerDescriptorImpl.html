26  package agave.internal;
27  
28  import java.lang.reflect.Method;
29  
30  import agave.CompletesWorkflow;
31  import agave.InitiatesWorkflow;
32  import agave.ResumesWorkflow;
33  
34  /**
35   * A descriptor that serves as the configuration for the building of handlers and routing requests 
36   * into them.
37   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
38   */
39   public class HandlerDescriptorImpl implements HandlerDescriptor {
40      
41      private URIPattern pattern;
42      private Class<?> handlerClass;
43      private Class<?> formClass;
44      private Method handlerMethod;
45      private boolean initiatesWorkflow;
46      private boolean completesWorkflow;
47      private String workflowName;
48  
49      public HandlerDescriptorImpl(HandlerIdentifier identifier) throws ClassNotFoundException {
50          pattern = new URIPatternImpl(identifier.getUri());
51          handlerClass = Class.forName(identifier.getClassName());
52          locateAnnotatedHandlerMethods(identifier);
53      }
54  
55      /**
56       * Locates annotated methods on a handler class.  This method will find the annotated method
57       * that is identified by the supplied {@code HandlerIdentifier} and then proceed to find any 
58       * workflow-related annotations ({@code InitiatesWorkflow}, {@code ResumesWorkflow},
59       * {@code CompletesWorkflow}.
60       * @param identifier the {@code HandlerIdentifier} that was created while scanning for a handler method
61       */
62      public void locateAnnotatedHandlerMethods(HandlerIdentifier identifier) {
63          for (Method method : handlerClass.getMethods()) {
64              if (identifier.getMethodName().equals(method.getName())) {
65                  handlerMethod = method;
66                  if (handlerMethod.getParameterTypes().length > 1) {
67                      formClass = handlerMethod.getParameterTypes()[1];
68                  }
69                  
70                  if (method.getAnnotation(InitiatesWorkflow.class) != null) {
71                      initiatesWorkflow = true;
72                      completesWorkflow = false;
73                      workflowName = method.getAnnotation(InitiatesWorkflow.class).value();
74                  } else if (method.getAnnotation(CompletesWorkflow.class) != null) {
75                      initiatesWorkflow = false;
76                      completesWorkflow = true;
77                      workflowName = method.getAnnotation(CompletesWorkflow.class).value();
78                  } else if (method.getAnnotation(ResumesWorkflow.class) != null) {
79                      initiatesWorkflow = false;
80                      completesWorkflow = false;
81                      workflowName = method.getAnnotation(ResumesWorkflow.class).value();
82                  }
83              }
84          }
85      }
86      
87      public URIPattern getPattern() {
88          return pattern;
89      }
90      
91      public Class<?> getHandlerClass() {
92          return handlerClass;
93      }
94  
95      public Class<?> getFormClass() {
96          return formClass;
97      }
98      
99      public Method getHandlerMethod() {
100         return handlerMethod;
101     }
102 
103     public boolean initiatesWorkflow() {
104         return initiatesWorkflow;
105     }
106 
107     public boolean completesWorkflow() {
108         return completesWorkflow;
109     }
110 
111     public String getWorkflowName() {
112         return workflowName;
113     }
114 
115     public int compareTo(HandlerDescriptor that) {
116         return getPattern().compareTo(that.getPattern());
117     }
118     
119     @Override
120     public boolean equals(Object that) {
121         if (this == that) {
122             return true;
123         }
124         if (!(that instanceof HandlerDescriptor)) {
125             return false;
126         }
127         HandlerDescriptor desc = (HandlerDescriptor)that;
128         return this.getPattern().equals(desc.getPattern());
129     }
130 
131     @Override
132     public int hashCode() {
133         return pattern.hashCode() + handlerClass.hashCode() + handlerMethod.hashCode();
134     }
135 
136     public boolean matches(String uri) {
137         return pattern.matches(uri);
138     }
