30  import javax.servlet.http.HttpServletRequest;

32  /**
33   * The {@code URIPattern} is an object constructed around the the argument 
34   * to the {@code @HandlesRequestsTo} annotation, and it indicates which 
35   * URL the handler method will field, based on the URL constructed by 
36   * combining the protocol, port, context, and URI.
37   *
38   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
39   */
40  public interface URIPattern extends Comparable<URIPattern> {
41      
42      public String getPattern();
43  
44      /**
45       * Determines whether the supplied URI string matches the pattern that this
46       * {@code URIPattern} encapsulates. The URI string supplied as an argument
47       * must start with a forward slash ('/'). The URI string is normalized with
48       * URI.normalize() from the Java API, then compared against the stored
49       * pattern where wildcards and replacement variables help determine the
50       * match. Replacement variables look like <code>${someVar}</code> and are
51       * taken as an automatic match. A single asterisk represents a wildcard
52       * match where the supplied token matches automatically as well. A double
53       * asterisk matches multiple tokens until the next token in the pattern is
54       * matched against the URI.
55       *
56       * @param uri the URI string
57       * @return true if the URI matches this pattern
58       */
59      public boolean matches(String uri);
60  
61      /**
62       * Normalizes the URI string so that .. and . are properly handled and condensed.
63       *
64       * @param uriStr the URI string to normalize
65       * @return the normalized URI string
66       */
67      public String normalizeURI(String uriStr);
68  
69      /**
70       * Indicates whether this {@code URIPattern}'s normalized internal
71       * pattern matches that of the supplied object's.
72       */
73      public boolean equals(Object obj);
74      public int hashCode();
75  
76      /**
77       * Compares two {@code URIPattern}s for greater specificity. A more
78       * specific {@code URIPattern} should always be sorted before a more generic
79       * one.
80       * @param that the {@code URIPattern} to compare against
81       * @return -1 if this {@code URIPattern} is more specific, 0 if they are
82       *         equal in specificity and 1 if that {@code URIPattern} is more
83       *         specific
84       */
85      public int compareTo(URIPattern that);
86  
87      public Map<String, String> getParameterMap(HttpServletRequest request);
88  
89  }
