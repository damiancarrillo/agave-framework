71  /**
72   * <p>
73   * Scans the {@code /WEB-INF/classes} directory of a deployed context for any configured 
74   * handlers, builds an internal representation of all handler methods, and then 
75   * forwards HTTP requests to the handlers if they match the requested URI.  See the
76   * {@link #doFilter(ServletRequest, ServletResponse, FilterChain)} method for a 
77   * description of the primary function of this filter.
78   * </p>
79   * 
80   * <p>
81   * This filter supports three {@code init-param}s:
82   * 
83   * <ul>
84   *   <li id="classesDirectory">{@code classesDirectory} - A directory created by calling {@code new File()} on the value of this parameter.
85   *   The default value that is used when this parameter is not present is {@code /WEB-INF/classes}.</li>
86   *   <li id="lifecycleHooks">{@code lifecycleHooks} - See {@link LifecycleHooks} for more information.</li>
87   *   <li id="instanceCreator">{@code instanceCreator} - See {@link InstanceCreator} for more information.</li>
88   * </ul>
89   * </p>
90   * 
91   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
92   * @see <a href="#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain)"><code>AgaveFilter.doFilter()</code></a>
93   * @see HandlerDescriptor
94   */

109      * An alternate way of providing the {@link LifecycleHooks} implementation is to override this method.  
110      * The default behavior is to read the <a href="#lifecycleHooks">lifecycleHooks</a> {@code init-param} and use the supplied 
111      * value to instantiate a {@link LifecycleHooks} instance, or use a stub implementation if this parameter
112      * is not supplied.
113      * 
114      * @param config the configuration delivered to this filter
115      * @return a {@link LifecycleHooks} instance
116      * 
117      * @throws ClassNotFoundException if the class named by the {@code lifecycleHooks} initialization parameter can not be found
118      * @throws InstantiationException if the class named by the {@code lifecycleHooks} initialization parameter can not be instantiated
119      * @throws IllegalAccessException if this filter is denied access to the class named by the {@code lifecycleHooks} initialization parameter value
120      */
121     protected LifecycleHooks provideLifecycleHooks(FilterConfig config) 
122         throws ClassNotFoundException, InstantiationException, IllegalAccessException {
123         LifecycleHooks hooks = null;
124         
125         String lifecycleHooksParameter = config.getInitParameter("lifecycleHooks");
126         if (lifecycleHooksParameter != null) {
127             hooks = (LifecycleHooks)Class.forName(lifecycleHooksParameter).newInstance();
128             LOGGER.info("Using lifecycle hooks: " + hooks.getClass().getName());
129         } else {
130             hooks = new DefaultLifecycleHooks();
131         }
132         
133         return hooks;
134     }

166      * An alternate way of providing a class directory to scan for handlers is to override this method.  This is primarily here for testing 
167      * situations (especially when running mvn jetty:run).  The default behavior is to read the 
168      * <a href="#classesDirectory">classesDirectory</a> {@code init-param} and use the value supplied to create a

170      * will be {@code /WEB-INF/classes}.
171      * 
172      * @param config the configuration delivered to this filter
173      * @return a {@code File} representing the named class directory
174      * 
175      * @throws ClassNotFoundException if the class named by the {@code classesDirectory} initialization parameter can not be found
176      * @throws InstantiationException if the class named by the {@code classesDirectory} initialization parameter can not be instantiated
177      * @throws IllegalAccessException if this filter is denied access to the class named by the {@code instanceCreator} initialization parameter value
178      */
179     protected File provideClassesDirectory(FilterConfig config)
180         throws ClassNotFoundException, InstantiationException, IllegalAccessException {
181         File classesDirectory = null;
182         
183         if (config.getInitParameter("classesDirectory") != null) {
184             classesDirectory = new File(config.getInitParameter("classesDirectory"));
185             classesDirectoryProvided = true;
186         } else {
187             classesDirectory = new File(config.getServletContext().getRealPath("/WEB-INF/classes"));
188         }
189         
190         return classesDirectory;

245     /**
246      * <p>
247      *   Handles the routing of HTTP requests through the framework. The algorithm
248      *   used internally is as follows:
249      * </p>
250      * 
251      * <ol>
252      *   <li>
253      *     {@link agave.InstanceCreator#createFormInstance Instantiates a
254      *     form if necessary}.
255      *     <ol>
256      *       <li>
257      *         {@link RequestParameterFormPopulator Populates 
258      *         request parameters} if necessary, leveraging any {@link agave.conversion.Converter}s named
259      *         with the {@link ConvertWith} annotation on mutator arguments.  Note that URI parameters
260      *         will override request parameters if they are similarly named. 
261      *       </li>
262      *       <li>
263      *         {@link URIParameterFormPopulator Populates 
264      *         URI parameters} if necessary, leveraging any {@link agave.conversion.Converter}s named
265      *         with the {@link ConvertWith} annotation on mutator arguments.
266      *       </li>
267      *     </ol>
268      *   </li>
269      *   <li>
270      *     {@link agave.InstanceCreator#createHandlerInstance
271      *      Instantiates a handler}.
272      *   </li>
273      *   <li>
274      *     Populates a new {@link HandlerContext}, or look up a previously created
275      *     one if this handler method participates in a workflow.
276      *   </li>
277      *   <li>
278      *     Invokes the handler method with the populated {@link HandlerContext} and the 
279      *     form instance (if applicable).
280      *   </li>
281      * </ol>
282      * 
283      * <p>
284      *   When one of the two supported encoding methods is selected, then this
285      *   filter will field the HTTP request that was made and will prevent future
286      *   execution of the filter chain. If an unsupported encoding type is
287      *   requested or if a handler is not configured for the requested URI, this
288      *   filter will simply continue with execution of the filter chain. The two
289      *   encoding types that are supported by this method are:
290      * </p>
291      * 
292      * <ul>
293      *   <li><a href="http://www.w3.org/TR/html401/interact/forms.html#h-17.13.4.1">
294      *     application/x-www-form-urlencoded</a></li>
295      *   <li><a href="http://www.w3.org/TR/html401/interact/forms.html#h-17.13.4.2">
296      *     multipart/form-data</a></li>
297      * </ul>
298      * 
299      * @param req the Servlet request object; it will be cast to an {@code HttpServletRequest}
300      * @param resp the Servlet response object; it will be cast to an {@code HttpServletResponse}
301      * @param chain the filter chain this filter is a member of
302      * @throws IOException if an I/O error occurs
303      * @throws ServletException if a Servlet error occurs
304      * 
305      * @see <a href="http://www.w3.org/TR/html401/interact/forms.html#h-17.13.4">W3C Form Encoding Types</a>
306      * @see agave.HandlesRequestsTo
307      * @see agave.ConvertWith
308      */
309     public final void doFilter(ServletRequest req, ServletResponse resp, FilterChain chain)
310             throws IOException, ServletException {
311         ServletContext servletContext = config.getServletContext();
312         HttpServletRequest request = (HttpServletRequest) req;
313         HttpServletResponse response = (HttpServletResponse) resp;
314 
315         HandlerDescriptor descriptor = handlerRegistry.findMatch(request);
316         if (descriptor != null) {
317             HttpSession session = request.getSession(true);
318             HandlerContext handlerContext =
319                 new HandlerContext(servletContext, request, response, session);
320 
321             lifecycleHooks.beforeFilteringRequest(descriptor, handlerContext);
322             
323             LOGGER.fine(request.getServletPath() + " -> " + descriptor.getHandlerClass().getName() + "#" + 
324                     descriptor.getHandlerMethod() + "()");
325 
326             if (MultipartRequestImpl.isMultipart(request)) {
327                 request = new MultipartRequestImpl(request);
328             }
329 
330             Object formInstance = null;
331             
332             if (descriptor.getWorkflowName() != null && !descriptor.initiatesWorkflow()) {
333                 formInstance = session.getAttribute(descriptor.getWorkflowName() + WORKFLOW_FORM_SUFFIX);
334             }
335             
336             if (formInstance == null) {
337                 formInstance = instanceFactory.createFormInstance(descriptor);
338             }
339             
340             if (formInstance != null) {
341                 if (descriptor.initiatesWorkflow()) {
342                     session.setAttribute(descriptor.getWorkflowName() + WORKFLOW_FORM_SUFFIX, formInstance);
343                 }
344                 
345                 lifecycleHooks.beforeHandlingRequest(descriptor, formInstance, handlerContext);
346                 
347                 try {
348                     FormPopulator formPopulator = new RequestParameterFormPopulator(request);
349                     formPopulator.populate(formInstance);
350                     if (MultipartRequestImpl.isMultipart(request)) {
351                         formPopulator = new RequestPartFormPopulator((MultipartRequest)request);
352                         formPopulator.populate(formInstance);
353                     }
354                     formPopulator = new URIParameterFormPopulator(request, descriptor);
355                     formPopulator.populate(formInstance);
356                 } catch (NoSuchMethodException ex) {
357                     throw new FormException(ex);
358                 } catch (IllegalAccessException ex) {
359                     throw new FormException(ex);
360                 } catch (InvocationTargetException ex) {
361                     throw new FormException(ex.getCause());
362                 } catch (InstantiationException ex) {
363                     throw new FormException(ex);
364                 }
365                 
366                 lifecycleHooks.afterInitializingForm(descriptor, formInstance, handlerContext);
367             }
368 
369             Object handlerInstance = null;
370             
371             if (descriptor.getWorkflowName() != null && !descriptor.initiatesWorkflow()) {
372                 handlerInstance = session.getAttribute(descriptor.getWorkflowName() + WORKFLOW_HANDLER_SUFFIX);
373             }
374             
375             if (handlerInstance == null) {
376                 handlerInstance = instanceFactory.createHandlerInstance(descriptor);
377             }
378             
379             if (descriptor.initiatesWorkflow()) {
380                 session.setAttribute(descriptor.getWorkflowName() + WORKFLOW_HANDLER_SUFFIX, handlerInstance);
381             }
382             
383             lifecycleHooks.beforeHandlingRequest(descriptor, handlerInstance, handlerContext);
384             Object result = null;
385             
386             try {
387                 if (formInstance != null) {
388                     if (descriptor.getHandlerMethod().getReturnType() != null) {
389                         result = descriptor.getHandlerMethod().invoke(handlerInstance, handlerContext, formInstance);
390                     } else {
391                         descriptor.getHandlerMethod().invoke(handlerInstance, handlerContext, formInstance);
392                     }
393                 } else {
394                     if (descriptor.getHandlerMethod().getReturnType() != null) {
395                         result = descriptor.getHandlerMethod().invoke(handlerInstance, handlerContext);
396                     } else {
397                         descriptor.getHandlerMethod().invoke(handlerInstance, handlerContext);
398                     }
399                 }
400             } catch (InvocationTargetException ex) {
401                 if (ex.getCause() instanceof AgaveException) {
402                     throw (AgaveException) ex.getCause();
403                 } else if (ex.getCause() instanceof IOException) {
404                     throw (IOException) ex.getCause();
405                 } else if (ex.getCause() instanceof RuntimeException) {
406                     throw (RuntimeException) ex.getCause();
407                 } else {
408                     throw new HandlerException(ex.getMessage(), ex.getCause());
409                 }
410             } catch (IllegalAccessException ex) {
411                 throw new HandlerException(descriptor, ex);
412             }
413 
414             if (descriptor.completesWorkflow()) {
415                 session.removeAttribute(descriptor.getWorkflowName() + WORKFLOW_HANDLER_SUFFIX);
416                 session.removeAttribute(descriptor.getWorkflowName() + WORKFLOW_FORM_SUFFIX);
417             }
418             
419             if (result != null && !response.isCommitted()) {
420                 URI uri = null;
421                 boolean redirect = false;
422                 
423                 if (result instanceof DestinationImpl) {
424                     Destination destination = (Destination)result;
425                     
426                     lifecycleHooks.afterHandlingRequest(descriptor, handlerInstance, destination, handlerContext);
427                     
428                     try {
429                         uri = new URI(null, destination.encode(config.getServletContext()), null);
430                         if (destination.getRedirect() == null) {
431                             if ("POST".equalsIgnoreCase(request.getMethod())) {
432                                 redirect = true;
433                             }
434                         } else {
435                             redirect = destination.getRedirect();
436                         }
437                     } catch (URISyntaxException ex) {
438                         throw new DestinationException(ex.getMessage(), ex.getCause());
439                     }
440                 } else if (result instanceof URI) {
441                     uri = (URI)result;
442                     
443                     lifecycleHooks.afterHandlingRequest(descriptor, handlerInstance, uri, handlerContext);
444                     
445                     redirect = true;
446                 } else {
447                     throw new DestinationException(descriptor);
448                 }
449                 
450                 if (redirect) {
451                     String location = uri.toASCIIString();
452                     if (location.startsWith("/")) {
453                         location = request.getContextPath() + location;
454                     }
455                     response.sendRedirect(location);
456                 } else {
457                     request.getRequestDispatcher(uri.toASCIIString()).forward(request, response);
458                 }
459             } else {
460                 lifecycleHooks.afterHandlingRequest(descriptor, handlerInstance, handlerContext);
461             }
462         } else {
463             chain.doFilter(req, resp);
464         }
465     }

467     /**
468      * Scans the supplied directory for handlers. Handlers in turn are inspected
469      * and have a {@link agave.internal.HandlerDescriptor HandlerDescriptor}
470      * generated for them which then gets registered in the
471      * {@link agave.internal.HandlerRegistry HandlerRegistry} as handlers are
472      * found.
473      * 
474      * @param root the root directory to scan files for, typically {@code /WEB-INF/classes}
475      */
476     protected void scanClassesDirForHandlers(File root) 
477     throws FileNotFoundException, IOException, ClassNotFoundException, ServletException {
478         if (root != null && root.canRead()) {
479             for (File node : root.listFiles()) {
480                 if (node.isDirectory()) {
481                     scanClassesDirForHandlers(node);
482                 } else if (node.isFile() && node.getName().endsWith(".class")) {
483                     lifecycleHooks.beforeHandlerIsDiscovered(node);
484                    
485                     FileInputStream nodeIn = new FileInputStream(node); 
486                     try {
487                         ClassReader classReader = new ClassReader(nodeIn);
488                         Collection<HandlerIdentifier> handlerIdentifiers = new ArrayList<HandlerIdentifier>();
489                         classReader.accept(new HandlerScanner(handlerIdentifiers), ClassReader.SKIP_CODE);
490                         
491                         for (HandlerIdentifier handlerIdentifier : handlerIdentifiers) {
492                             HandlerDescriptor descriptor = new HandlerDescriptorImpl(handlerIdentifier);
493                             descriptor.locateAnnotatedHandlerMethods(handlerIdentifier);
494                             handlerRegistry.addDescriptor(descriptor);
495                             
496                             lifecycleHooks.afterHandlerIsDiscovered(descriptor, config.getServletContext());
497                         }
498                     } finally {
499                         nodeIn.close();
500                     }
501                 }
502             }
503         }
504     }
