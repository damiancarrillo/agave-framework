69      @Before
70      public void setup() throws Exception {
71          filterChain = context.mock(FilterChain.class);
72          filterConfig = context.mock(FilterConfig.class);
73          request = context.mock(HttpServletRequest.class);
74          response = context.mock(HttpServletResponse.class);
75          servletContext = context.mock(ServletContext.class);
76          requestDispatcher = context.mock(RequestDispatcher.class);
77          session = context.mock(HttpSession.class);
78      }
79      
80      protected void emulateServletContainer(final Map<String, String[]> parameters)
81          throws URISyntaxException {
82  
83          final URL rootUrl = getClass().getClassLoader().getResource("agave");
84          final String realPath = new File(rootUrl.toURI()).getAbsolutePath();
85  
86          context.checking(new Expectations() {{
87              allowing(servletContext).getRealPath("/WEB-INF/classes"); will(returnValue(realPath));
88              allowing(filterConfig).getServletContext(); will(returnValue(servletContext));
89              allowing(filterConfig).getInitParameter("lifecycleHooks"); will(returnValue(null));
90              allowing(filterConfig).getInitParameter("classesDirectory"); will(returnValue(null));
91              allowing(filterConfig).getInitParameter("instanceCreator"); will(returnValue(null));
92              allowing(request).getLocale(); will(returnValue(Locale.ENGLISH));
93              allowing(request).getSession(true); will(returnValue(session));
94              allowing(request).getParameterMap(); will(returnValue(parameters));
95              allowing(request).getParameterNames(); will(returnValue(new Vector<String>(parameters.keySet()).elements()));
96              for (String parameter : parameters.keySet()) {
97                  allowing(request).getParameterValues(parameter); will(returnValue(parameters.get(parameter)));
98              }
99          }});
100     }
101     
102     protected AgaveFilter scanRoot() throws Exception {
103         URL rootUrl = getClass().getClassLoader().getResource("agave");
104         Assert.assertNotNull(rootUrl);
105         File root = new File(rootUrl.toURI());
106         Assert.assertNotNull(root);
107 
108         AgaveFilter filter = new AgaveFilter();
109         
110         Logger agaveFilterLogger = LogManager.getLogManager().getLogger(AgaveFilter.class.getName());
111         if (agaveFilterLogger != null) {
112             agaveFilterLogger.setLevel(Level.OFF);
113         }
114         
115         emulateServletContainer(new HashMap<String, String[]>());
116         
117         filter.init(filterConfig);
118         filter.setHandlerRegistry(new HandlerRegistryImpl());
119         filter.scanClassesDirForHandlers(root);
120         return filter;
121     }
122     
123     protected AgaveFilter createSilentAgaveFilter() throws Exception {
124         AgaveFilter filter = new AgaveFilter();
125         
126         Logger agaveFilterLogger = LogManager.getLogManager().getLogger(AgaveFilter.class.getName());
127         if (agaveFilterLogger != null) {
128             agaveFilterLogger.setLevel(Level.OFF);
129         }
130         
131         return filter;
132     }
