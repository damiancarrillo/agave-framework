26  package agave.sample;
27  
28  import agave.HandlerContext;
29  import java.io.IOException;
30  
31  import javax.servlet.ServletException;
32  
33  import agave.HandlesRequestsTo;
34  
35  public class SampleHandler {
36      @HandlesRequestsTo("/login")
37      public void login(HandlerContext context, LoginForm loginForm) throws ServletException, IOException {
38          if ("damian".equals(loginForm.getUsername()) && "password".equals(loginForm.getPassword())) {
39              context.getRequest().setAttribute("loggedIn", Boolean.TRUE);
40          } else {
41              context.getRequest().setAttribute("loggedIn", Boolean.FALSE);
42          }
43          context.getResponse().setStatus(400);
44      }
45  
46      @HandlesRequestsTo("/aliased")
47      public void aliased(HandlerContext context, AliasedForm aliasedForm) throws ServletException, IOException {
48      }
49  
50      @HandlesRequestsTo("/uri-params/${username}/${password}/")
51      public void uriParams(HandlerContext context, LoginForm loginForm) throws ServletException, IOException {
52          context.getRequest().setAttribute("username", loginForm.getUsername());
53          context.getRequest().setAttribute("password", loginForm.getPassword());
54      }
55  
56      @HandlesRequestsTo("/throws/nullPointerException")
57      public void throwsNullPointerException(HandlerContext context, LoginForm loginForm) throws ServletException, IOException {
58          throw new NullPointerException();
59      }
60  
61      @HandlesRequestsTo("/throws/ioException")
62      public void throwsIOException(HandlerContext context, LoginForm loginForm) throws ServletException, IOException {
63          throw new IOException();
64      }
65      
66      @HandlesRequestsTo("/lacks/form")
67      public void lacksForm(HandlerContext context) throws ServletException, IOException {
68          context.getRequest().setAttribute("noErrors", Boolean.TRUE);
69      }
70  
71  }
