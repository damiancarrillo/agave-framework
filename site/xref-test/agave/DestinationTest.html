39  public class DestinationTest {
40      
41      Mockery context = new Mockery();
42      ServletContext servletContext;
43      
44      @Before
45      public void setup() throws Exception {
46          servletContext = context.mock(ServletContext.class);
47      }

49      @Test
50      public void testConstructorWithOnlyPathArgument() throws Exception {
51          Destination dest = Destinations.create("/some/resource");
52          Assert.assertNotNull(dest);
53          Assert.assertNull(dest.getRedirect());
54      }
55      
56      @Test
57      public void testConstructorWithPathAndRedirectArguments() throws Exception {
58          Destination dest = Destinations.redirect("/some/resource");
59          Assert.assertNotNull(dest);
60          Assert.assertNotNull(dest.getRedirect());
61          Assert.assertTrue(dest.getRedirect());
62      }
63      
64      @Test(expected = IllegalArgumentException.class)
65      public void testConstructorWithInvalidPath() throws Exception {
66          Destination dest = Destinations.create("some/resource");
67          Assert.assertNotNull(dest);
68      }
69      
70      @Test
71      public void testAddingParameters() throws Exception {
72          Destination dest = Destinations.create("/some/resource");
73          dest.addParameter("dog", "woof");
74          dest.addParameter("dog", "bark");
75          dest.addParameter("cat", "meow");
76          dest.addParameter("cat", "purr");
77          
78          Assert.assertNotNull(dest.getParameters());
79          Assert.assertNotNull(dest.getParameters().get("dog"));
80          Assert.assertNotNull(dest.getParameters().get("cat"));
81          Assert.assertTrue(dest.getParameters().get("dog").contains("woof"));
82          Assert.assertTrue(dest.getParameters().get("dog").contains("bark"));
83          Assert.assertTrue(dest.getParameters().get("cat").contains("meow"));
84          Assert.assertTrue(dest.getParameters().get("cat").contains("purr"));
85      }
86      
87      @Test
88      public void testEncode() throws Exception { 
89          Destination dest = Destinations.create("/some/resource");
90          dest.addParameter("dog", "woof");
91          dest.addParameter("dog", "bark");
92          dest.addParameter("cat", "meow");
93          dest.addParameter("cat", "purr");
94          
95          context.checking(new Expectations() {{
96              allowing(servletContext).getContextPath(); will(returnValue("/app"));
97          }});
98          
99          Assert.assertEquals("/some/resource?cat=meow&cat=purr&dog=bark&dog=woof", dest.encode(servletContext));
100         dest.addParameter("bird", "chirp&chirp");
101         Assert.assertEquals("/some/resource?bird=chirp&amp;chirp&cat=meow&cat=purr&dog=bark&dog=woof", dest.encode(servletContext));
102     }
